$(document).ready(function(){ 

 var currLocation = window.location.href;
 var wh = $(window).height()-70;

 $("#sec_0").css({height:wh});
 $("#sec_1").css({height:wh});

 if(currLocation == "https://2016.igem.org/Team:Valencia_UPV/Team"){
    var wh = $(window).height()-130;
    var ww = $(window).width();

    $("#title_team").height(wh);
    $("#title_team").width(ww);

    $(".col-md-2").css({height:450});
    $(".col-md-2").css({width:360});
    $(".container-fluid").css({padding:0});
    $(".container-fluid").css({margin:0});
    $(".row").css({margin:0});

   $("#first").hover(function(){
      $("#first_inside").fadeIn("fast");
      $("#img_cont_team").css({background: linear-gradient(rgba(0, 0, 0, 0.7), rgba(0, 0, 0, 0.7))});
    },function(){
      $("#first_inside").fadeOut("fast");
    });

    if(ww <= 1400){
      $("#first").removeClass("col-md-offset-2");
      $("#first").addClass("col-md-offset-1");

      $(".con").css({"margin-right":"1.5em"});
    }
 }

});
