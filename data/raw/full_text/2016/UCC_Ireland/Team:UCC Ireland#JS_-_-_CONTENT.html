$( document ).ready(function() { 

 $(window).scroll(function() {
     var scroll = $(window).scrollTop();
     var os = $('#myCarousel').offset().top;
     var ht = $('#myCarousel').height();
     // scrolled down
     if(scroll > 60){
         $('nav').css('cssText', 'opacity: .9 !important');
         $('nav a').css('cssText', 'color: black !important');
         $('.navbar-nav > .active > a').css('cssText', 'color: orange !important');
     }
     //scrolled up
     if(scroll <= 60){
         $('nav').css('cssText', 'opacity: .6 !important');
         $('nav a').css('cssText', 'color: white !important');

         $('.navbar-nav > .active > a').css('cssText', 'color: orange !important');
     }

 });

});
