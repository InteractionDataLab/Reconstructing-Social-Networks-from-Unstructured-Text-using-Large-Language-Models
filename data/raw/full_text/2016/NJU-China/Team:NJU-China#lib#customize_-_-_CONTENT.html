// 轮播图居中 setTimeout(function(){ centerIt(); },200); $(window).resize(function() { centerIt(); }); function centerIt() { var height1 = $("#myCarousel").height(); var height2 = $(".col-md-7").height(); var width1 = $("#myCarousel").width(); var width2 = $("#myCarousel").parent().width(); var hasFloat = $("#myCarousel").parent().css("float"); var topDistance = (height2-height1)/2+"px"; var leftDistance = (width2-width1)/2+"px"; $("#myCarousel").css("left", leftDistance); // 轮播图div父元素.col-md-5有width，有float，说明处于大屏幕状态，排在右侧 if(hasFloat!="none") { $(".col-md-7").css("border-right", '2px dashed #EE8848'); $("#myCarousel").css("top", 'unset'); if(height1<height2) $("#myCarousel").css("top", topDistance); } // 轮播图div父元素.col-md-5无width，无float，说明处于小屏幕状态，排在下面 else { $("#myCarousel").css("top", "unset"); $("#myCarousel").css("left", leftDistance); $(".col-md-7").css("border-right", 'unset'); } }
