jQuery(document).ready(function($) { $(".scroll").click(function(event){ event.preventDefault(); $('html,body').animate({scrollTop:$(this.hash).offset().top},1000); }); });

(function() { 

   "use strict";
   $("html").niceScroll({styler:"fb",cursorcolor:"#79302a", cursorwidth: '6', cursorborderradius: '10px', background: '#666', spacebarenabled:false, cursorborder: '0',  zindex: '1000'});

   $(".scrollbar1").niceScroll({styler:"fb",cursorcolor:"#79302a", cursorwidth: '6', cursorborderradius: '10px',autohidemode: 'false', background: '#F1F1F1', spacebarenabled:false, cursorborder: '0'});

   $(".scrollbar1").getNiceScroll();
   if ($('body').hasClass('scrollbar1-collapsed')) {
       $(".scrollbar1").getNiceScroll().hide();
   }

})(jQuery);
