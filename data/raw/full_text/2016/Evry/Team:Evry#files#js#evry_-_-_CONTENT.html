$(document).ready(function() { 

   var s = $("#page_wrap");
   var pos = s.position();
   $(window).scroll(function() {
       var windowpos = $(window).scrollTop();
       if (windowpos >= 67) {
           s.addClass("scrolled");
       } else {
           s.removeClass("scrolled");
       }
   });

}); $(window).resize(function() { 

   if ($(window).width() <= 1015) {
       $('#page_wrap').addClass('sscreen');
   } else {
       $('#page_wrap').removeClass('sscreen');
   }

});

function offsetAnchor() { 

   if(location.hash.length !== 0) {
       window.scrollTo(window.scrollX, window.scrollY - 80);
   }

}

// This will capture hash changes while on the page $(window).on("hashchange", offsetAnchor);

// This is here so that when you enter the page with a hash, // it can provide the offset in that case too. Having a timeout // seems necessary to allow the browser to jump to the anchor first. window.setTimeout(offsetAnchor, 1); // The delay of 1 is arbitrary and may not always work right (although it did in my testing).

$(document).ready(function() { 

   if ($(window).width() <= 1015) {
       $('#page_wrap').addClass('sscreen');
   }

}); $(function() { 

   $('#returntotop').click(function() {
       if (location.pathname.replace(/^\//, ) == this.pathname
           .replace(/^\//, ) && location.hostname == this.hostname
       ) {
           var target = $(this.hash);
           target = target.length ? target : $('[name=' + this
               .hash.slice(1) + ']');
           if (target.length) {
               $('html, body').animate({
                   scrollTop: target.offset().top
               }, 500);
               return false;
           }
       }
   })
   $('.cat_menu.dropdown').click(function() {
       $(this).siblings().removeClass('hovered').children().next()
           .slideUp('fast');
       $('.subsub_menu').hide(1);
       $('.shown').removeClass('shown');
       $(this).toggleClass('hovered').children('.sub_menu').slideToggle(
           'fast');
       return false;
   });
   $('html').click(function() {
       if ($('.sub_menu').is(':visible')) {
           $('.sub_menu').slideUp('fast');
           $('.hovered').removeClass('hovered');
       }
   });
   $('.show_more').click(function() {
       if ($('#show_more').is(':visible')) {
           $('.show_more a').replaceWith(
               '<a href="#">Show menu</a>'
           );
       } else {
           $('.show_more a').replaceWith(
               '<a href="#">Hide menu</a>'
           );
       }
       $('#show_more').slideToggle('fast');
       return false;
   });
   $(window).scroll(function() {
       if ($('#show_more').is(':visible')) {
           $('#show_more').slideUp('fast');
       }
   });
   $(window).resize(function() {
       if ($('#show_more').is(':visible')) {
           $('#show_more').slideUp('fast');
       }
   });
   $('#navbar #nav ul li.cat_menu ul li a').click(function() {

window.location.href = $(this).attr('href'); 

   });
