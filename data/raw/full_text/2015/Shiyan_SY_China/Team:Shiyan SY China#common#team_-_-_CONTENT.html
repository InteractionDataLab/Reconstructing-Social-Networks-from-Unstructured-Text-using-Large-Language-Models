$(document).ready(function(){

$(".section ul li .rsp").hide();

$(".section ul li").hover(function(){ $(this).find(".rsp").stop().fadeTo(500,0.8) $(this).find(".text").stop().animate({left:'0'}, {duration: 500}) },function(){ $(this).find(".rsp").stop().fadeTo(500,0) $(this).find(".text").stop().animate({left:'318'}, {duration: "fast"}) $(this).find(".text").animate({left:'-318'}, {duration: 0}) });

});

$(function () {

var cont_left = $("#container").position().left; 

   $("UL.templates LI").hover(function () {
       // hover in
       $(this).css("z-index", 2);
       animateB($(this).find("A"));
       animateB($(this).find("IMG"));
   }, function () {
       // hover out
       $(this).css("z-index", 0);
       reversAnimationB($(this).find("A"));
       reversAnimationB($(this).find("IMG"));
   });

   function animateB(obj) {
       $(obj).animate({
           height: "180",
           width: "165",
           left: "-=10",
           top: "-=10"
       }, "1000");
   }

   function reversAnimationB(obj) {
       $(obj).animate({
           height: "160",
           width: "145",
           left: "+=10",
           top: "+=10"
       }, "1000");
   }

   $("#container a").each(function (index) {
       var left = (index * 0) + cont_left;
       $(this).css("left", left + "px");
   });

});
