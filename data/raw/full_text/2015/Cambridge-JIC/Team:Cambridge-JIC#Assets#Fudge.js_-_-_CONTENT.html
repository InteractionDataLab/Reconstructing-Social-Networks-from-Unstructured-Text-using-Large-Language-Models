function hndl(me) { 

   if($(me).parent().children("div").is(':visible')) {
       $(".navsub_m").hide()
   } else {
       $(".navsub_m").hide()
       $(me).parent().children("div").show()
   }

}

String.prototype.endsWith = function(suffix) { 

   return ~this.indexOf(suffix, this.length - suffix.length);

};
