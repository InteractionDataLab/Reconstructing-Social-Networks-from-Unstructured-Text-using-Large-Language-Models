$(window).scroll(function() { 

 var wScroll = $(this).scrollTop();
 var numberOfCards = $('.card').length;

 for (i = 0; i < numberOfCards; i++) {

   if (wScroll > $('.card').eq(i).offset().top - ($(window).height() / 1.4)) {
     $('.card').eq(i).addClass('visible');
   }
 }

 var bScroll = $(document).height() - wScroll - $(window).height() - $('footer').height();
 if (bScroll < 300) {
   $('img.scrolldown').addClass('hidden');
 }

 //  circle parallaxing effect
 var elementOffsetTop = [];
 var belowSpace = [];
 var translate = [];
 var n = $('.circle').length;

 for (i = 0; i < n; i++) {
   elementOffsetTop[i] = $('.circle').eq(i).offset().top;
   belowSpace[i] = $(window).height() - elementOffsetTop[i] + wScroll;
   if (belowSpace[i] < 0) {
     backToTop(i);
   } else if (belowSpace[i] > ($(window).height() + parseInt($('.circle').eq(i).css('height'))) && (wScroll + $(window).height()) < ($(document).height() - $('footer').outerHeight())) {
     backToBottom(i);
   };

   if ($('.circle').eq(i).hasClass('small')) {
     translate[i] = offset[i] + wScroll / 2;
   } else if ($('.circle').eq(i).hasClass('normal')) {
     translate[i] = offset[i] + wScroll / 1.5;
   } else if ($('.circle').eq(i).hasClass('large')) {
     translate[i] = offset[i] + wScroll / 1.2;
   } else if ($('.circle').eq(i).hasClass('halflarge')) {
     translate[i] = offset[i] + wScroll / 1.2;
   }

   $('.circle').eq(i).css({
     'transform': 'translateY(' + translate[i] + 'px)'
   });
 };

 function backToTop(n) {
   offset[n] = offset[n] - $(window).height() - parseInt($('.circle').eq(n).css('height'));
 }

 function backToBottom(n) {
   offset[n] = offset[n] + $(window).height() + parseInt($('.circle').eq(n).css('height'));
 }

});

var offset = new Array(10).fill(0);
