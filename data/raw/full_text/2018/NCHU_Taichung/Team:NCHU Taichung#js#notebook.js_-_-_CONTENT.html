     case indexPrev:
       indexMonth -= 1;
       break;
   }
 }

}

 switch (true) {
   case next:
     $(buttonId).on("click", function () {
       if (indexMonth >= 2) {
         $(mainClass).css({
           left: "+=100%"
         });
         $(monthClass).css({
           left: "+=100%"
         });
         indexMonth -= 1;
       }
       return indexMonth;
     });
     break;
   case prev:
     $(buttonId).on("click", function () {
       if (indexMonth <= 11) {
         $(mainClass).css({
           left: "-=100%"
         });
         $(monthClass).css({
           left: "-=100%"
         });
         indexMonth += 1;
       }
       return indexMonth;
     });
     break;
 }

}
