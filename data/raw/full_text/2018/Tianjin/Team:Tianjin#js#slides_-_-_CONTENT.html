$(document).ready(function(){ 

   var slideIndex = 1;
   
 $(".wrap").each(function(j){
 showDivs(slideIndex);
 $(".wrap:eq("+j+") .display-right").click(function(){plusDivs(1,j);});
 $(".wrap:eq("+j+") .display-left").click(function(){plusDivs(-1,j);});

 $(".wrap:eq("+j+") .cr").each(function(e){
 $(".wrap:eq("+j+") .cr:eq("+e+")").click(function(){
   currentSlide(e);
   cractive(e);
   $(this).addClass("active");
 })})
 function currentSlide(n){
   showDivs(slideIndex = (n+1), 1)
 }
 function cractive(n){
   $(".wrap:eq("+j+") .cr").each(function(ee){if(ee!=n){$(this).removeClass("active")};})
 }
 function plusDivs(n){
   showDivs(slideIndex += n, n);
 }
 function showDivs(n,k){
   var i;
   var x = $(".wrap:eq("+j+") .mySlides");
   if (n > x.length) {slideIndex = 1}    
   if (n < 1) {slideIndex = x.length}

   if(k>0){right(slideIndex);}
   if(k<0){left(slideIndex);}
   function right(n){
     $(".wrap:eq("+j+") .mySlides").each(function(e){
     $(this).css("display","none");
     $(this).removeClass("slideRInAni slideROutAni slideLInAni slideLOutAni");
   })
     $(".wrap:eq("+j+") .mySlides:eq("+(slideIndex-2)+")").addClass("slideROutAni");
     $(".wrap:eq("+j+") .mySlides:eq("+(slideIndex-1)+")").addClass("slideRInAni");

     $(".wrap:eq("+j+") .cr").each(function(e){
     $(this).removeClass("active");
   })
     $(".wrap:eq("+j+") .cr:eq("+(slideIndex-1)+")").addClass("active");
   }
   function left(n){
     $(".wrap:eq("+j+") .mySlides").each(function(e){
     $(this).css("display","none");
     $(this).removeClass("slideRInAni slideROutAni slideLInAni slideLOutAni");
   })
     $(".wrap:eq("+j+") .mySlides:eq("+(slideIndex)+")").addClass("slideLOutAni");
     $(".wrap:eq("+j+") .mySlides:eq("+(slideIndex-1)+")").addClass("slideLInAni");

     $(".wrap:eq("+j+") .cr").each(function(e){
     $(this).removeClass("active");
   })
     $(".wrap:eq("+j+") .cr:eq("+(slideIndex-1)+")").addClass("active");
   }

   $(".wrap:eq("+j+") .mySlides:eq("+(slideIndex-1)+")").css("display","block");
 }
 })

})
