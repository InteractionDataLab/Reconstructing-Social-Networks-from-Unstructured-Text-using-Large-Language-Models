$(window).ready(function () { 

 //Credit to Yaphi Berhanu for this idea and code segments

 //global vars
 var hp_slides = document.querySelectorAll("#hp_slider .hp_slides");
 var hp_slideCount = 0;
 var hp_interval;

 //bind Event-Listener
 var prevButton = document.getElementById("hp_slider-previous");
 prevButton.addEventListener("click", prevSlide);
 var nextButton = document.getElementById("hp_slider-next");
 nextButton.addEventListener("click", nextSlide);

 //setInterval for Auto-Show
 resetInterval();

 function goToSlide(n) {
   hp_slides[hp_slideCount].className = 'hp_slides';
   hp_slideCount = (n + hp_slides.length) % hp_slides.length;
   hp_slides[hp_slideCount].className = 'hp_slides hp_slides-is_visible';
 }

 function nextSlide() {
   goToSlide(hp_slideCount + 1);
   resetInterval();
 }

 function prevSlide() {
   goToSlide(hp_slideCount - 1);
   resetInterval();
 }

 function resetInterval() {
   clearInterval(hp_interval);
   hp_interval = setInterval(function () {
     goToSlide(hp_slideCount + 1)
   }, 4976); //weil janek es so wollte
 }

 //JQuery zum Anzeigen der Bilder in großem Format
 $(".hp_slides img").click(function () {
   //currently selected image and paragraph that describes it
   var image = $(this);
   var paragraph = image.parent().children("p");

   //overlay-Objects inside a Jquery-Funktion
   var overlayImage = $("#hp_slider-overlay_img img");
   var overlayParagraph = $("#hp_slider-overlay_img p");
   var overlayDiv = $("#hp_slider-overlay_img");

   //change Attribute and Description previous to showing the image
   overlayImage.attr("src", image.attr("src"));
   overlayParagraph.html(paragraph.html());
   overlayDiv.show(500);
 });

 $("#hp_slider-exit_button").click(function () {
   $("#hp_slider-overlay_img").hide(500);
 })

});
