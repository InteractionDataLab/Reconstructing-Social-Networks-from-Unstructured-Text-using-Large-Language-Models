%This files aims at defining all the needed parameters. Please refer to the

%array in the modeling page for the meanings.

global B

global b

global R

global D

global O

global g

global Ncell

global Next

Ncell = 1;

Next = Ncell*8 + 1;

B(1,1)=50;

B(2,1)=1200;

B(3,1)=600;

B(4,1)=100;

b(1,1)=0;

b(2,1)=250;

b(3,1)=300;

b(4,1)=350;

R(1)=20;

B(5,1)=0.01;

D(1,1)=2;

D(2,1)=2;

O(1)=1;

g(1) = 1;

g(2) = 1;

g(3) = 1;

g(4) = 1;

g(5) = 1;

g(6) = 1;

g(8) = 1;

g(Next) = 1;

for k=2:Ncell

B(1,k)=noise(B(1,1));

B(2,k)=noise(B(2,1));

B(3,k)=noise(B(3,1));

B(4,k)=noise(B(4,1));

b(1,k)=noise(b(1,1));

b(2,k)=noise(b(2,1));

b(3,k)=noise(b(3,1));

b(4,k)=noise(b(4,1));

R(k)=noise(R(1));

B(5,k)=noise(B(5,1));

D(1,k)=noise(D(1,1));

D(2,k)=noise(D(2,1));

O(k)=noise(O(1));

g(1+8*(k-1)) = noise(g(1));

g(2+8*(k-1)) = noise(g(2));

g(3+8*(k-1)) = noise(g(3));

g(4+8*(k-1)) = noise(g(4));

g(5+8*(k-1)) = noise(g(5));

g(6+8*(k-1)) = noise(g(6));

g(8+8*(k-1)) = noise(g(8));

end
