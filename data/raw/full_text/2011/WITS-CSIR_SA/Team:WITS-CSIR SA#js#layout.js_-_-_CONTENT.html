function layout() { 

   $('#target').css({
       width: window.innerWidth + 'px',
       height: window.innerHeight + 'px'
   })

   var container = $('div.layout');
   container.layout({ resize: false, type: 'border', vgap: 0, hgap: 0 });

   $('div.banner').css({ background: CreateLinearGradient('#627AAD', '#3C5792', 'vertical', $('div.banner').height() + 20, 1) });

   if ($('#hidStart').length > 0 && $('#hidStart').val() == 'true') {
       $('body').css({ background: CreateLinearGradient('#eeeeff', '#ccccff', 'vertical', $('body').height() + 30, 1) });
   }
   else {
       $('body').css({ background: 'white' });
   }

   $('.lightblue').css({ background: CreateLinearGradient('#eeeeff', '#ccccff', 'vertical', $('body').height() + 30, 1) });

   var isStart = $('body').hasClass('start');

   var offset = $('div.center').offset();
   //    $('h1').eq(0).css({
   //        position: 'absolute',
   //        left: (offset.left - (!isStart ? $('.west').width() + 25 : 0)) + 'px'
   //    });

   //    $('h1').eq(1).css({
   //        position: 'absolute',
   //        left: (offset.left + 960 - 64 - (!isStart ? $('.west').width() + 25 : 0)) + 'px'
   //    });

   $('div.menu').css({
       paddingLeft: (offset.left + 5 - (!isStart ? $('.west').width() + 25 : 0)) + 'px'
   });

   var logo = $('#logo');
   var igemLogo = $('#igemLogo');

   var offset = $('div.north').offset();
   logo.css({ position: 'absolute', left: (offset.left + 5) + 'px' });

   igemLogo.css({ position: 'absolute', left: (offset.left + $('div.north').width() - igemLogo.width()) + 'px' });

   $('div.searchBox').css({
       position: 'absolute',
       left: ((window.innerWidth - 960) / 2 + 635) + 'px',
       top: '0px'
   });

   FormatButtons();

}

$(function () { 

   $.fling('subscribe', 'ready', function () {
       // Setup page root layout
       layout();

       $('.north').resizable({
           handles: 's',
           stop: layout,
           resize: layout
       });

       $('.south').resizable({
           handles: 'n',
           stop: layout,
           resize: layout
       });

       $('.east').resizable({
           handles: 'w',
           stop: layout,
           resize: layout
       });

       $('.west').resizable({
           handles: 'e',
           stop: layout,
           resize: layout
       });

       $(window).resize(layout);

       layout();
   });

});
