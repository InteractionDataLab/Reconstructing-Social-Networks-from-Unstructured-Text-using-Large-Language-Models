$(document).ready(function(){ 

     function pinDrop(pin){
              if($('#teampin'+pin).hasClass('animating') == false){
                  $('#teampin'+pin).addClass('animating').animate({'top':'-=150px','opacity':'0'},1,function(){$('#teampin'+pin).animate({top:'+=150px',opacity:1},200,function(){$('#teampin'+pin).removeClass('animating');});});
                  }
              if($('#teampin'+pin+'shadow').hasClass('animating') == false){
                  $('#teampin'+pin+'shadow').addClass('animating').animate({'top':'-=50px','left':'+=50px'},1,function(){$('#teampin'+pin+'shadow').animate({top:'+=50px',left:'-=50'},200,function(){$('#teampin'+pin+'shadow').removeClass('animating');});});
                  }
             }

    $('#click').click(function(){ pinDrop(1);});

});
