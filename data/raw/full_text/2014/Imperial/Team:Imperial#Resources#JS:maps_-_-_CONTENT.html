$(function () { 

   $.ajax({
       type: 'GET',
       url: 'http://jsonp.guffa.com/Proxy.ashx?url=https://2014.igem.org/Team:Imperial/Resources/json:CitableDocuments?action=raw&ctype=text/json',
       dataType: 'jsonp',
       success: function (data) {
           // Initiate the chart
           $('#CitableDocuments').highcharts('Map', {

               title: {
                   text: 'Biotechnology Citable Documents (1996-2013)'
               },

               mapNavigation: {
                   enabled: true,
                   buttonOptions: {
                       verticalAlign: 'bottom'
                   }
               },

               colorAxis: {
                   min: 100,
                   max: 90000,
                   type: 'logarithmic',
                   stops: [
               [0, '#3060cf'],
               [0.5, '#fffbbc'],
               [0.9, '#fdcc70']
                   ]
               },

               series: [{
                   data: data,
                   mapData: Highcharts.maps['custom/world-robinson-highres'],
                   joinBy: ['iso-a2', 'code'],
                   name: 'Biotechnology  Citable Documents',
                   states: {
                       hover: {
                           color: '#BADA55'
                       }
                   },
                   tooltip: {
                       valueSuffix: 
                   }
           }]
           });
       }
   });

});
