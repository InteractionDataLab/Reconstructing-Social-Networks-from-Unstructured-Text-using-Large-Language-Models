body, .cbp-fbscroller, .cbp-fbscroller { height: 100%; }

/* The nav is fixed on the right side and we center it by translating it 50% (we don't know it's height so we can't use the negative margin trick) */ .cbp-fbscroller > nav { position: fixed; z-index: 9999; right: 5%; top: 50%; width: 26px; -webkit-transform: translateY(-50%); -moz-transform: translateY(-50%); -ms-transform: translateY(-50%); transform: translateY(-50%); }

.cbp-fbscroller > nav a { display: block; position: relative; z-index: 9999; color: transparent; width: 15px; height: 15px; outline: none; margin: 16px 0; border-radius: 50%; border: 4px solid #ddd; }

.no-touch .cbp-fbscroller > nav a:hover { background: rgba(255,255,255,0.6); }

.cbp-fbscroller > nav a.cbp-fbcurrent { background: #eee; }

/* background-attachment does the trick */ .cbp-fbscroller section { position: relative; background-position: top center; background-repeat: no-repeat; background-size: cover; }

.no-touch .cbp-fbscroller section { background-attachment: fixed; }
