sliderInt = 1; sliderNext = 2; count = 1;

$(document).ready(function () { 

   $('#slider>div#1').fadeIn(300);
   count = $("#slider>div").size();
   startSlider();

});

function startSlider() { 

   loop = setInterval(function () {

       if (sliderNext > count) {
           sliderNext = 1;
           sliderInt = 1;
       }

       for (i = 1; i <= count; i++) {
           var astring = "#thumbnails> ul > li > a#p" + i.toString();
           $(astring).css('opacity', 0.8);
       }

       $("#thumbnails> ul > li > a#p" + sliderNext).css('opacity', 1);

       $("#slider>div").fadeOut(0);
       $("#slider>div#" + sliderNext).fadeIn(300);

       sliderInt = sliderNext;
       sliderNext = sliderNext + 1;

   }, 5000);

}

function stopLoop() { 

  window.clearInterval(loop);

}

function showSlide2(id) { 

   stopLoop();
   if (id > count) {
       id = 1;
   } else if (id < 1) {
       id = count;
   }

   for (i = 1; i <= count; i++) {
       var astring = "#thumbnails> ul > li > a#p" + i.toString();
       $(astring).css('opacity', 0.8);
   }

   $("#thumbnails> ul > li > a#p" + id).css('opacity', 1);

   $("#slider>div").fadeOut(0);
   $("#slider>div#" + id).fadeIn(300);

}

//1. set all thumbnails to 0.8 //2. set current thumbnail to 1

$("#slider > div").hover( 

   function () {
       stopLoop();
   },
   function () {
       startSlider();
   }

);

$("#thumbnails > ul > li > a").hover( 

   function () {
       //$(this).css('opacity', 1);
       sliderInt = parseInt($(this).attr('href').substring(1));
       showSlide2(sliderInt);
       sliderNext = sliderInt + 1;
   }
   ,
   function () {
       //$(this).css('opacity', 0.8);
       startSlider();
   }

);
