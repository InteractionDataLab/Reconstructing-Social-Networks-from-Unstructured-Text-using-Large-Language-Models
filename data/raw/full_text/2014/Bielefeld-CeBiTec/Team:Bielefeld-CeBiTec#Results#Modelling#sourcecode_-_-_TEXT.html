%kivD v5 = (kcat(5) * conc(5) * y(5)) / (km(5) + y(5)); %adhA v6 = (kcat(6) * conc(6) * y(6)) / (km(6) + y(6)); %pyruvatase v7 = (kcat(6) * conc(6) * y(6)) / (km(6) + y(6)); %pyruvatase %differentiation systems dy_dt(1) = -v1;
