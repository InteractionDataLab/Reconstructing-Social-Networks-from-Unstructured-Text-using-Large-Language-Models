/* Panel Wrapper */ .mb-slider, .mb-scroll { width: 100%; height: 100%; overflow: hidden; margin: 0 auto; padding: 0; position: relative; left: 0; top: 0;

/***(>'-')> Control Panel Font size here <('-'<)***/ font-size: 18px; }

/* active slider border highlight */ .mb-active-slider { border-color: #333; }

/*** Slider panel ***/ .mb-slider .mb-panel { margin: 0; padding: 20px; display: block; cursor: pointer; float: left; list-style: none; }
