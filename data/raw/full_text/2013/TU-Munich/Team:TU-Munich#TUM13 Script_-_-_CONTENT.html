$.urlParam = function(name){ 

   var results = new RegExp('[\\?&]' + name + '=([^&#]*)').exec(window.location.href);
   if (results==null){
      return null;
   }
   else{
      while (results[1].indexOf('+') != -1) {
          results[1] = results[1].replace('+', ' ');
      }
      return results[1] || 0;
   }

};
