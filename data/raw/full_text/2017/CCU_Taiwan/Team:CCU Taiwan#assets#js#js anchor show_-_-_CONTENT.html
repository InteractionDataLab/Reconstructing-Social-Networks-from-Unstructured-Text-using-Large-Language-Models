$(function(){ 　$(window).load(function(){ 　　$(window).bind('scroll resize', function(){ 　　var $this = $(this); 　　var $this_Top=$this.scrollTop();

　　if($this_Top > 400 && $this_Top < $(document).height()-750){ $('#toc_show').fadeIn('fast');

　　　} 　　　　else{ $('#toc_show').fadeOut('fast');　　

　　　 } 　　}).scroll(); 　}); });

$(function(){ 

 //Scroll event
 $(window).scroll(function(){
   var scrolled = $(window).scrollTop();
   if (scrolled > 200) $('.go-top').fadeIn('slow');
   if (scrolled < 200) $('.go-top').fadeOut('slow');
 });
 
 //Click event
 $('.fixed').click(function () {
   $("html, body").animate({ scrollTop: "0" },500);
 });

});
