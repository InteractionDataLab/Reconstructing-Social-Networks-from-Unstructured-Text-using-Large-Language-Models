end %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% k2_AFT.m %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% function [ k1 ] = k2_AFT( ScFv2, AFT, ScFv2_AFT ) %FUNK1 Summary of this function goes here % Detailed explanation goes here k1 = ScFv2_AFT / (ScFv2 * AFT); end %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% afun.m %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% function [ A ] = afun( AFT, k1, k2, k3 ) %AFUN Summary of this function goes here % Detailed explanation goes here A = k3 * k1 * AFT/(k1 * AFT + 1) * k2 * AFT/(k2 * AFT + 1); end %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% prfun.m %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% function [ Pr ] = prfun( t, c_5, c_6, c_7, k_5, k_7 ) %PRFUN Summary of this function goes here % Detailed explanation goes here Pr = c_5 * exp( -k_7 * t) + c_6 + c_7 * exp(-k_5 * t);
